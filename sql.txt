CREATE TABLE publishers (
    pub_id   CHAR(3) NOT NULL,
    pub_name VARCHAR(20) NOT NULL,
    year_est VARCHAR(4),
    city     VARCHAR(15) NOT NULL,
    state    CHAR(2)             ,
    country  VARCHAR(15) NOT NULL,
  CONSTRAINT pk_publishers PRIMARY KEY (pub_id)
  );
